# rogsoft ASUSWRT KaiPlus Split Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Split ASUSWRT/Koolshare KaiPlus into `rogsoft_KaiPlus` while keeping `rogsoft_LinkEase` independent and able to reverse-proxy KaiPlus when installed.

**Architecture:** `rogsoft_LinkEase` owns only LinkEase Desktop and `apptunnel-client`; it detects an installed KaiPlus plugin and passes `KAIPLUS_PROXY_TARGET` to Desktop. `rogsoft_KaiPlus` owns KaiPlus files, dbus keys, service lifecycle, firewall port `8189`, and direct `/apps/kaiplus/` access.

**Tech Stack:** POSIX shell for Koolshare plugin lifecycle scripts, ASUSWRT dbus/nvram/iptables conventions, Python `unittest` for contract tests, Python build scripts for tarball packaging.

## Global Constraints

- LinkEase and KaiPlus must remain independently installable and removable.
- KaiPlus default direct URL is `http://<router-ip>:8189/apps/kaiplus/`.
- LinkEase Desktop reverse-proxy URL is `http://<router-ip>:<LinkEase Desktop port>/apps/kaiplus/`.
- KaiPlus default port is `8189`.
- KaiPlus default base path is `/apps/kaiplus/`.
- KaiPlus ASUSWRT system role is `asusgo`.
- LinkEase must export `KAIPLUS_ENABLED=0`.
- LinkEase must not export `KAIPLUS_BIN`, `KAIPLUS_STATIC_DIR`, `KAIPLUS_DEFAULTS_DIR`, `KAIPLUS_HOME`, `KAIPLUS_ADDR`, or `KAIPLUS_BASE_PATH`.
- LinkEase must not copy, chmod, start, stop, or uninstall `/koolshare/kaiplus` or `/koolshare/bin/kaiplus_bin`.
- LinkEase build must stage only `linkease-desktop` and `apptunnel-client`.
- KaiPlus build must stage `kaiplus_bin`, `kaiplus_workspace_tool`, `www`, and `defaults`.
- User data under external `.linkease_data` must not be deleted by install or uninstall scripts.
- First ASUSWRT full runtime remains arm64/aarch64 only.

- [ ] **Step 3: Add proxy target resolver**

```sh
normalize_kaiplus_port(){
	case "$1" in
		''|*[!0-9]*) echo 8189; return 0 ;;
	esac
	if [ "$1" -ge 1 ] 2>/dev/null && [ "$1" -le 65535 ] 2>/dev/null; then
		echo "$1"
	else
		echo 8189
	fi
}

resolve_kaiplus_proxy_target(){
	KAIPLUS_PROXY_TARGET=""
	[ -x /koolshare/scripts/kaiplus_config.sh ] || return 0
	[ -x /koolshare/bin/kaiplus_bin ] || return 0

	kaiplus_port="$(dbus get kaiplus_port 2>/dev/null)"
	kaiplus_port="$(normalize_kaiplus_port "$kaiplus_port")"
	KAIPLUS_PROXY_TARGET="http://127.0.0.1:${kaiplus_port}"
	export KAIPLUS_PROXY_TARGET="${KAIPLUS_PROXY_TARGET}"
}
```

- [ ] **Step 1: Create `build.py`**

Use this complete Python script:

```python
#!/usr/bin/env python3

import argparse
import hashlib
import json
import shutil
import stat
import tarfile
from pathlib import Path


parent_path = Path(__file__).resolve().parent
EXPECTED_MODULE = "kaiplus"


def md5sum(path):
    return hashlib.md5(Path(path).read_bytes()).hexdigest()


def make_executable(path):
    path = Path(path)
    path.chmod(path.stat().st_mode | stat.S_IXUSR | stat.S_IXGRP | stat.S_IXOTH)


def copy_tree(src, dst):
    src = Path(src)
    dst = Path(dst)
    if dst.exists():
        shutil.rmtree(dst)
    shutil.copytree(src, dst)


def load_config(root):
    config_path = Path(root) / "config.json.js"
    if not config_path.is_file():
        raise FileNotFoundError(f"missing config file: {config_path}")
    return json.loads(config_path.read_text(encoding="utf-8"))


def write_config(root, conf):
    config_path = Path(root) / "config.json.js"
    config_path.write_text(json.dumps(conf, sort_keys=True, indent=4, ensure_ascii=False) + "\n", encoding="utf-8")


def validate_module_name(module):
    module = str(module or "").strip()
    if (
        not module
        or module in (".", "..")
        or "/" in module
        or "\\" in module
        or Path(module).is_absolute()
        or Path(module).name != module
    ):
        raise ValueError("module must be a simple directory name")
    return module


def stage_artifacts(module_dir, artifact_dir):
    module_dir = Path(module_dir)
    artifact_dir = Path(artifact_dir)

    binary_src = artifact_dir / "bin" / "kaiplus_bin"
    helper_src = artifact_dir / "helpers" / "kaiplus_workspace_tool"
    www_src = artifact_dir / "www"
    defaults_src = artifact_dir / "defaults"

    for required in (binary_src, helper_src):
        if not required.is_file():
            raise FileNotFoundError(f"missing required artifact: {required}")
    for required in (www_src, defaults_src):
        if not required.is_dir():
            raise FileNotFoundError(f"missing required artifact directory: {required}")

    bin_dir = module_dir / "bin"
    helper_dir = module_dir / "helpers"
    bin_dir.mkdir(parents=True, exist_ok=True)
    helper_dir.mkdir(parents=True, exist_ok=True)

    shutil.copy2(binary_src, bin_dir / "kaiplus_bin")
    shutil.copy2(helper_src, helper_dir / "kaiplus_workspace_tool")
    make_executable(bin_dir / "kaiplus_bin")
    make_executable(helper_dir / "kaiplus_workspace_tool")

    copy_tree(www_src, module_dir / "www")
    copy_tree(defaults_src, module_dir / "defaults")
    for script in (module_dir / "defaults").glob("**/scripts/*"):
        if script.is_file():
            make_executable(script)


def package_module(root, module):
    root = Path(root)
    package_path = root / f"{module}.tar.gz"
    if package_path.exists():
        package_path.unlink()
    with tarfile.open(package_path, "w:gz") as tf:
        tf.add(root / module, arcname=module)
    return package_path


def build_module(root=None, artifact_dir=None):
    root = Path(root or parent_path)
    conf = load_config(root)
    module = validate_module_name(conf.get("module"))
    if module != EXPECTED_MODULE:
        raise ValueError(f"module must be {EXPECTED_MODULE}")
    module_dir = root / module
    if not module_dir.is_dir():
        raise FileNotFoundError(f"missing module directory: {module_dir}")
    if artifact_dir is None:
        raise ValueError("artifact_dir is required")

    stage_artifacts(module_dir, artifact_dir)
    (module_dir / "version").write_text(str(conf.get("version", "0.1.0")) + "\n", encoding="utf-8")
    package_path = package_module(root, module)
    conf["md5"] = md5sum(package_path)
    write_config(root, conf)
    return conf


def main():
    parser = argparse.ArgumentParser(description="Build KaiPlus Asus plugin package")
    parser.add_argument("--artifact-dir", required=True, help="Directory containing bin/kaiplus_bin, helpers/kaiplus_workspace_tool, www, and defaults")
    args = parser.parse_args()
    conf = build_module(artifact_dir=args.artifact_dir)
    print(f"build done {conf['module']}.tar.gz md5={conf['md5']}")


if __name__ == "__main__":
    main()
```
